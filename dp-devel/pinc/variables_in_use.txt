doctype.inc
     $docType = standard <DOCTYPE> header

metabaseurl.inc
     $baseURL = site's basic URL to the home page root    

cookiecheck.inc
     $cookieC  class instance use as $cookieSet->setCookie() | $cookieSet->deleteCookie()

     $cookieID = the username as pulled from the pgname cookie variable
